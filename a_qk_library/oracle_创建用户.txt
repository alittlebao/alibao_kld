创建临时表空间（ORA-02199: 缺失 DATAFILE/TEMPFILE 子句）
create temporary tablespace bpm_temp  
tempfile 'E:\app\Alibao\oradata\bpm\bpm_temp.dbf' 
size 50m  
autoextend on  
next 50m maxsize 20480m  
extent management local;

创建表空间
create tablespace bpm  
logging  
datafile 'E:\app\Alibao\oradata\bpm\bpm_data.dbf' 
size 50m  
autoextend on  
next 50m maxsize 20480m  
extent management local; 

创建bpm用户并赋予bpm表空间
create user bpm identified by 1234 account unlock default tablespace bpm temporary tablespace bpm_temp;

用户授权 
grant connect,resource,dba to bpm;

赋予用户所有权限
grant all privileges to bpm;
